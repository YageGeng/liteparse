Usage

        
import init, { LiteParse } from "@llamaindex/liteparse-wasm";

// Initialize the WASM module
await init();

// Create one parser instance and reuse it across files
// Enable layout detection by setting layoutEnabled and tuning thresholds
const parser = new LiteParse({
  outputFormat: "json",
  layoutEnabled: true,
  layoutConfidenceThreshold: 0.25,
  layoutIouThreshold: 0.45,
  layoutImageSize: 1280
});

// Parse PDF bytes into CLI-compatible JSON or Markdown
const bytes = new Uint8Array(await file.arrayBuffer());
const json = await parser.parseJson(bytes);
const markdown = await parser.parseMarkdown(bytes);
const layoutImages = await parser.layoutScreenshots(bytes);

console.log(json); // same schema as the CLI JSON output
